When y <- x is run, R does not immediately copy the data — both x and y point to the same vector in memory. Once y is modified, R creates a separate copy for y so that x remains unchanged. This is called copy-on-modify: R only copies data when it has to, which keeps results independent while saving memory.

  x1 <- x[1]

}

rosenbrock_grad <- function(x) {
  x1 <- x[1]
  x2 <- x[2]
  
  c(
    -2 * (1 - x1) - 400 * x1 * (x2 - x1^2),
    200 * (x2 - x1^2)
  )
}

(c)

gradient_descent <- function(init, grad_fn, fn,
                             learning_rate = 0.001,
                             max_iter = 10000,
                             tol = 1e-6) {
  x <- init
  
  for (iter in 1:max_iter) {
    grad <- grad_fn(x)
    
    # stop if Euclidean norm of gradient is small enough
    if (sqrt(sum(grad^2)) < tol) {
      return(list(
        solution = x,
        value = fn(x),
        iterations = iter - 1,
        converged = TRUE
      ))
    }
    
    x <- x - learning_rate * grad
  }
  
  # if max_iter reached before convergence
  list(
    solution = x,
    value = fn(x),
    iterations = max_iter,
    converged = FALSE
  )
}

run_simulation <- function(n_samples, n_reps, dist_type) {
  # Initialize storage for this condition
  estimates <- matrix(
    NA,
    nrow = n_reps,
    ncol = 3,
    dimnames = list(NULL, c("mean", "median", "trimmed"))
  )
  
  # Run n_reps replications
  for (k in 1:n_reps) {
    # Generate data under the specified distribution
    if (dist_type == "normal") {
      x <- rnorm(n_samples, mean = 0, sd = 1)
    } else if (dist_type == "t3") {
      x <- rt(n_samples, df = 3)
    } else if (dist_type == "contaminated") {
      # 90% from N(0,1), 10% from N(0,10)
      n_clean <- rbinom(1, n_samples, 0.9)
      x_clean <- rnorm(n_clean, mean = 0, sd = 1)
      x_cont <- rnorm(n_samples - n_clean, mean = 0, sd = sqrt(10))
      x <- c(x_clean, x_cont)
    } else {
      stop("dist_type must be 'normal', 't3', or 'contaminated'")
    }
    
    # Compute estimators
    estimates[k, "mean"] <- mean(x)
    estimates[k, "median"] <- median(x)
    estimates[k, "trimmed"] <- mean(x, trim = 0.1)
  }
  
  # Return results as a data frame
  return(as.data.frame(estimates))
}

(b)

summarize_metrics <- function(estimates_df, true_value = 0, dist_name) {
  summary_df <- data.frame(
    distribution = character(),
    estimator = character(),
    bias = numeric(),
    variance = numeric(),
    mse = numeric(),
    stringsAsFactors = FALSE
  )
  
  for (est in c("mean", "median", "trimmed")) {
    est_values <- estimates_df[[est]]
    
    bias_val <- mean(est_values) - true_value
    variance_val <- var(est_values)
    mse_val <- mean((est_values - true_value)^2)
    
    summary_df <- rbind(
      summary_df,
      data.frame(
        distribution = dist_name,
        estimator = est,
        bias = bias_val,
        variance = variance_val,
        mse = mse_val,
        stringsAsFactors = FALSE
      )
    )
  }
  
  return(summary_df)
}

bootstrap_mean <- function(x, B = 2000, ci_type = c("percentile", "basic", "normal")) {
  ci_type <- match.arg(ci_type)
  
  n <- length(x)
  theta_hat <- mean(x)
  
  # Store bootstrap means
  boot_stats <- numeric(B)
  
  # Monte Carlo bootstrap
  for (b in 1:B) {
    x_star <- sample(x, size = n, replace = TRUE)
    boot_stats[b] <- mean(x_star)
  }
  
  # Bootstrap standard error
  se_boot <- sd(boot_stats)
  
  # 95% CI
  alpha <- 0.05
  z <- qnorm(1 - alpha / 2)
  q_lower <- quantile(boot_stats, probs = alpha / 2)
  q_upper <- quantile(boot_stats, probs = 1 - alpha / 2)
  
  if (ci_type == "percentile") {
    ci <- unname(c(q_lower, q_upper))
  } else if (ci_type == "basic") {
    ci <- unname(c(2 * theta_hat - q_upper, 2 * theta_hat - q_lower))
  } else if (ci_type == "normal") {
    ci <- c(theta_hat - z * se_boot, theta_hat + z * se_boot)
  }
  
  return(list(
    estimate = theta_hat,
    se_boot = se_boot,
    ci = ci,
    boot_stats = boot_stats,
    ci_type = ci_type
  ))
}

The returns data are somewhat right-skewed because they were generated by combining a normal component with an exponential component. For skewed data, the percentile interval is often the most appropriate bootstrap CI because it directly uses the empirical quantiles of the bootstrap distribution and naturally reflects asymmetry. The normal interval may be less appropriate because it forces symmetry around the sample mean. The basic interval is also bootstrap-based, but for these data the percentile method is the most intuitive and usually the most reasonable choice.

bootstrap_ttest <- function(x, y, B = 2000) {
  n1 <- length(x)
  n2 <- length(y)
  
  # Observed difference in means
  obs_diff <- mean(y) - mean(x)
  
  # Enforce null hypothesis by centering both groups to common mean
  pooled_mean <- mean(c(x, y))
  x_null <- x - mean(x) + pooled_mean
  y_null <- y - mean(y) + pooled_mean
  
  # Bootstrap null distribution
  null_diffs <- numeric(B)
  
  for (b in 1:B) {
    x_star <- sample(x_null, size = n1, replace = TRUE)
    y_star <- sample(y_null, size = n2, replace = TRUE)
    null_diffs[b] <- mean(y_star) - mean(x_star)
  }
  
  # Two-sided bootstrap p-value
  p_boot <- mean(abs(null_diffs) >= abs(obs_diff))
  
  return(list(
    obs_diff = obs_diff,
    null_diffs = null_diffs,
    p_value = p_boot
  ))
}

The bootstrap test and the standard t-test should give similar results here because the data were generated from roughly normal distributions and the sample sizes are the same. In this case, the t-test is simpler and works well. The bootstrap approach is still helpful because it relies less on distribution assumptions. If the data were more skewed or had outliers, the bootstrap method would probably be more useful.
